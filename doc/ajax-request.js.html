define(['jquery'], function ($) {

    function toggleLoadingState(isOn) {
        var $spinner = $('.spinner');
        if (isOn) {
            if ($spinner.is(':visible') === false) {
                $spinner.show();
            }
        }
        else {
            if ($spinner.is(':visible') === true) {
                $spinner.hide();
            }
        }
    }
